Null

Nullは振幅0のGainである.

extern crate autd3;
use autd3::prelude::*;
#[allow(unused_variables)]
fn main()  {
let g = autd3::gain::Null::new();
}
#include "autd3.hpp"

auto g = autd3::gain::Null();

using AUTD3Sharp.Gain;

var g = new Null();

from pyautd3.gain import Null

g = Null()
